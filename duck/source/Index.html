Ext.define('App.view.albums.Index', {
    extend:'Ext.panel.Panel',
    alias:'widget.albums.Index',
    title:"Albums grid",
    layout:{
        type:'hbox',
        align:'stretch'
    },
    initComponent:function () {
        var me = this;
        me.items = [
            {

                hasDeleteItems:true,
                hasEditItems:true,

                dispatch:"albums/",

                store:'Albums',
                xtype:'AutoGrid'
            }
        ];
        me.callParent(arguments);
    },
    closable:true,
    dockedItems:[
        {
            xtype:'toolbar',
            dock:'top',
            items:[
                {
                    text:'Add',
                    handler:function () {
                        Ext.History.add('albums/add', true);
                    }
                }
            ]
        }
    ]
});
