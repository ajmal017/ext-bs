Ext.define('App.view.tracks.Index', {
    extend:'Ext.panel.Panel',
    alias:'widget.tracks.Index',
    title:'Tracks Grid',
    layout:{
        type:'hbox',
        align:'stretch'
    },
    initComponent:function () {
        var me = this;
        me.items = [
            {

                hasDeleteItems:true,
                hasEditItems:true,

                dispatch:"tracks/",

                titleField:'name',
                store:'Tracks',
                xtype:'AutoGrid'
            }
        ];
        me.callParent(arguments);
    },
    dockedItems:[
        {
            xtype:'toolbar',
            dock:'top',
            items:[
                {
                    text:'Add',
                    handler:function () {
                        Ext.History.add('tracks/add', true);
                    }
                }
            ]
        }
    ]
});
