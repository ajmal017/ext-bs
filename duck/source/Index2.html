Ext.define('App.view.artists.Index', {
    extend:'Ext.panel.Panel',
    alias:'widget.artists.Index',
    title:'Artists Grid',
    layout:{
        type:'hbox',
        align:'stretch'
    },
    initComponent:function () {
        var me = this;
        me.items = [
            {
                hasDeleteItems:true,
                hasEditItems:true,
                dispatch:"artists/",
                store:'Artists',
                xtype:'AutoGrid'
            }
        ];
        me.callParent(arguments);
    },
    dockedItems:[
        {
            xtype:'toolbar',
            dock:'top',
            items:[
                {
                    text:'Add',
                    handler:function () {
                        Ext.History.add('artists/add', true);
                    }
                }
            ]
        }
    ]
});
